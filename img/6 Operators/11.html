void main()
{
    char ch;
    system("cls");

    printf("Enter any character");

    if (ch >= 65 && ch <= 90)
    {
        printf("Upper case character");
    }
    else if (ch >= 97 && ch <= 122)
    {
        printf("lower case character");
    }
    else if (ch >= 48 && ch <= 57)
    {
        printf("Digits");
    }
    else
    {
        printf("Special character");
    }
}
